# Unreal Engine 5 - MyActor 프로젝트 정리

## 1️⃣ 프로젝트 개요
- C++ Actor 클래스 (`MyActor`)를 생성
- 랜덤 이동 + 회전 10회
- 매 이동 시 좌표와 Step 출력
- 50% 확률 이벤트 발생
- 최종 이동 거리 및 이벤트 횟수 출력

## 2️⃣ 클래스 구조

### MyActor.h

#pragma once
#include "CoreMinimal.h"
#include "GameFramework/Actor.h"
#include "MyActor.generated.h"

UCLASS()
class UN_REPORT_API AMyActor : public AActor
{
	GENERATED_BODY()

public:
	AMyActor();

protected:
	virtual void BeginPlay() override;

public:
	virtual void Tick(float DeltaTime) override;

private:
	void Move();
	void Turn();
	void TriggerEventWithProbability(float Probability);

	int MoveCount;
	float TotalDistance;
	int EventCount;

	FVector PreviousLocation;
};

###MyActor.cpp

#include "MyActor.h"
#include "Math/UnrealMathUtility.h"
#include "Engine/Engine.h"

AMyActor::AMyActor()
{
	PrimaryActorTick.bCanEverTick = true;

	MoveCount = 0;
	TotalDistance = 0.f;
	EventCount = 0;
	PreviousLocation = FVector(0.f, 50.f, 0.f);

	SetActorLocation(PreviousLocation);
}

void AMyActor::BeginPlay()
{
	Super::BeginPlay();
}

void AMyActor::Tick(float DeltaTime)
{
	Super::Tick(DeltaTime);

	if (MoveCount < 10)
	{
		Move();
		Turn();

		FString LocationStr = FString::Printf(TEXT("Step %d | Location: X=%.3f Y=%.3f Z=%.3f"), 
			MoveCount, PreviousLocation.X, PreviousLocation.Y, PreviousLocation.Z);

		if (GEngine)
			GEngine->AddOnScreenDebugMessage(-1, 5.f, FColor::Green, LocationStr);

		TriggerEventWithProbability(50.f);

		MoveCount++;

		if (MoveCount == 10)
		{
			FString Summary = FString::Printf(TEXT("Total Distance: %.2f | Event Count: %d"), TotalDistance, EventCount);
			if (GEngine)
				GEngine->AddOnScreenDebugMessage(-1, 10.f, FColor::Yellow, Summary);
		}
	}
}

void AMyActor::Move()
{
	FVector Offset(FMath::RandRange(-100.f, 100.f), FMath::RandRange(-100.f, 100.f), 0.f);
	FVector NewLocation = PreviousLocation + Offset;

	TotalDistance += FVector::Dist(PreviousLocation, NewLocation);

	PreviousLocation = NewLocation;

	SetActorLocation(NewLocation);
}

void AMyActor::Turn()
{
	FRotator Rotation(0.f, FMath::RandRange(0.f, 360.f), 0.f);
	AddActorLocalRotation(Rotation);
}

void AMyActor::TriggerEventWithProbability(float Probability)
{
	int32 RandomValue = FMath::RandRange(1, 100);
	if (RandomValue <= Probability)
	{
		EventCount++;
		if (GEngine)
			GEngine->AddOnScreenDebugMessage(-1, 5.f, FColor::Red, TEXT("Event Triggered!"));
	}
}
