#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream file("C:/git study/NBC_CH2_7/art.txt");

    // 파일을 찾지못하였을때
    if (!file.is_open()) {
        cout << "파일 불러오기 실패" << endl;
        return 1;
    }

    string line;

    // 파일내용출력
    while (getline(file, line)) {
        cout << line << '\n';
    }

    return 0;
}
